	function createCookie(){
	   setCookie("aaconsentEN","aaconsentEN", "365" );
	   document.getElementById('consent').style.display = "none";

	function setCookie(name,value,days) {
		var expires = "";
		if (days) {
		    var date = new Date();
		    date.setTime(date.getTime() + (days*24*60*60*1000));
		    expires = "; expires=" + date.toUTCString();
		}
		document.cookie = name + "=" + (value || "")  + expires + "; path=/";
	}
	
	function getCookie(name) {
	    var nameEQ = name + "=";
	    var ca = document.cookie.split(';');
	    for(var i=0;i < ca.length;i++) {
	        var c = ca[i];
	        while (c.charAt(0)==' ') 
	        	c = c.substring(1,c.length);
	        if (c.indexOf(nameEQ) == 0)
	        	return c.substring(nameEQ.length,c.length);
	    }
	    return null;
	}

<h1>Boko Haram kills 23 mourners after Nigeria funeral</h1>


	 

	
	





	<div class="article-default-image">
		<img src="https://vid.alarabiya.net/images/2019/07/28/dae3bd93-96cb-4399-a527-aed53bf93311/dae3bd93-96cb-4399-a527-aed53bf93311_16x9_788x442.jpg" alt="" />
		<div class="article-image-info">
			<div class="inner">
				This screen grab image taken on January 2, 2018 from a video released by Boko Haram shows their fighters during a Christmas Day attack on a military checkpoint in Molai village on the outskirts of the northeast Nigerian city of Maiduguri. (File photo: AFP)
			</div>
		</div>
	</div>

		<p>Boko Haram gunmen on Saturday killed 23 mourners in Borno state in Nigeria&rsquo;s restive northeast after they attended a funeral, local militia and residents said.</p><p class="hide-in-mobile">At around 1030 GMT, the attackers on three motorbikes opened fire on a group of men as they walked back from a funeral in Nganzai district near the state capital Maiduguri, local militia leader Bunu Bukar Mustapha said.</p><p class="hide-in-mobile">&ldquo;Our men recovered 23 dead bodies from the scene of the attack, which happened this morning,&rdquo; Mustapha told AFP.</p><p class="hide-in-mobile">The men were returning to Badu Kuluwu from nearby Goni Abachari village where they had attended funeral prayers for a relative, Mustapha said.</p><p class="hide-in-mobile">An official from the Nganzai district confirmed both the incident and death toll.</p><p class="hide-in-mobile">&ldquo;I received calls from Nganzai that Boko Haram killed 23 people near Badu Kuluwu this morning,&rdquo; said the official, who asked not to be identified for his own safety.</p><p class="hide-in-mobile">Local hunters and militia recovered the bodies after survivors returned to the village and alerted them, Nganzai resident Saleh Masida said.</p><p class="hide-in-mobile">Boko Haram fighters have been repeatedly attacking Nganzai district.</p><p class="hide-in-mobile">Boko Haram and its ISIS-supported ISWAP splinter have intensified attacks on civilian and military targets in recent months.</p><p class="hide-in-mobile">Late Thursday, Boko Haram fighters attacked a camp for displaced people outside Maiduguri, killing two residents and looting food supplies after burning a nearby military base.</p><p class="hide-in-mobile">The decade-long conflict has spilled over into neighboring Chad, Niger, and Cameroon, claiming around 27,000 lives and forcing more than two million people to flee their homes.</p>

        function isEmail(email) {
